        const plantModal = document.getElementById('plantModal');
        const closeModalButton = document.getElementById('closeModalButton');
        const modalPlantName = document.getElementById('modalPlantName');
        const modalPlantSpecies = document.getElementById('modalPlantSpecies');
        // const modalPlantImage = document.getElementById('modalPlantImage'); // Removed
        const modalPlantDescription = document.getElementById('modalPlantDescription');
        const modalPlantSize = document.getElementById('modalPlantSize');
        const modalPlantWater = document.getElementById('modalPlantWater');
        const modalPlantSun = document.getElementById('modalPlantSun');
        const modalPlantBloom = document.getElementById('modalPlantBloom');
        const modalPlantWildlife = document.getElementById('modalPlantWildlife');
        const modalPlantNotes = document.getElementById('modalPlantNotes');

        // Open Modal Function
        function openModal(plantId) {
            const plant = plantData[plantId];
            if (plant) {
                modalPlantName.textContent = plant.name;
                modalPlantSpecies.textContent = plant.species;
                // modalPlantImage.src = plant.imageModal; // Removed
                // modalPlantImage.alt = plant.name; // Removed
                modalPlantDescription.textContent = plant.description;
                modalPlantSize.textContent = plant.matureSize;
                modalPlantWater.textContent = plant.waterNeeds;
                modalPlantSun.textContent = plant.sunExposure;
                modalPlantBloom.textContent = plant.bloomTime;
                modalPlantWildlife.textContent = plant.wildlifeAttracted;
                modalPlantNotes.textContent = plant.notes;
                plantModal.classList.remove('hidden');
                plantModal.classList.add('flex'); 
            }
        }

        // Close Modal Function
        function closeModal() {
            plantModal.classList.add('hidden');
            plantModal.classList.remove('flex');
        }

        // Event Listeners for Plant Tiles
        document.querySelectorAll('.plant-tile').forEach(tile => {
            tile.addEventListener('click', () => {
                const plantId = tile.dataset.plantId;
                openModal(plantId);
            });
        });

        // Event Listener for Close Button
        closeModalButton.addEventListener('click', closeModal);

        // Event Listener for Clicking Outside Modal Content
        plantModal.addEventListener('click', (event) => {
            if (event.target === plantModal) { 
                closeModal();
            }
        });

        // Close modal with Escape key
        document.addEventListener('keydown', (event) => {
            if (event.key === 'Escape' && !plantModal.classList.contains('hidden')) {
                closeModal();
            }
        });
